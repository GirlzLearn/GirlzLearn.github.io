Richter Scale

val = [1.0, 5.0, 9.1, 9.2]

def conversion(richter):
  joules = 10 ** ((1.5 * richter) + 4.8)
  tnt = joules * (1 / (4.184 * 10 ** 9))
  return "{} \t {} \t {}".format(richter, joules, tnt)

print("Richter \t Joules \t TNT")
for num in val:
  print(conversion(num))

richter = float(input("\nEnter a Richter Scale Value: "))

joules = 10 ** ((1.5 * richter) + 4.8)
tnt = joules * (1 / (4.184 * 10 ** 9))

print("Your richter scale value: ", richter)
print("Equivalence in joules: ", joules)
print("Equivalence in tons of TNT: ", tnt)
