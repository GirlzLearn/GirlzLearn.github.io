Canoe Portage

rod = int(input("Enter the number of rods: "))
print("You entered {} rods".format(rod))

m = rod * 5.0292
furlong = rod / 40
mile = rod / (1609.34 / 5.0292)
feet = rod / (0.3048 / 5.0292)
minutes = (60 * mile) / 3.1

print("Conversions: ")
print("Meters: ", m)
print("Feet: ", feet)
print("Miles: ", mile)
print("Furlong: ", furlong)
print("Minutes to walk {} rods: {}".format(rod, minutes))
